export const defaultVersion = "12 (bookworm)";

export const GenDebianConfig = (version) => {
  const debianName = debianVersionMap[version] || "bookworm";
  const isOldStable = version === "10 (buster)";
  const isUnstable = version === "unstable (sid)";
  
  // unstable (sid) only has a single source
  if (isUnstable) {
    return `# Source mirrors are commented out by default to improve apt update speed. Uncomment if needed.
deb https://mirrors.jcut.edu.cn/debian/ sid main contrib non-free non-free-firmware
# deb-src https://mirrors.jcut.edu.cn/debian/ sid main contrib non-free non-free-firmware`;
  }
  
  // Older versions use archive path
  const securityPath = isOldStable ? `${debianName}/updates` : `${debianName}-security`;
  
  return `# Source mirrors are commented out by default to improve apt update speed. Uncomment if needed.
deb https://mirrors.jcut.edu.cn/debian/ ${debianName} main contrib non-free non-free-firmware
# deb-src https://mirrors.jcut.edu.cn/debian/ ${debianName} main contrib non-free non-free-firmware
deb https://mirrors.jcut.edu.cn/debian/ ${debianName}-updates main contrib non-free non-free-firmware
# deb-src https://mirrors.jcut.edu.cn/debian/ ${debianName}-updates main contrib non-free non-free-firmware
deb https://mirrors.jcut.edu.cn/debian/ ${debianName}-backports main contrib non-free non-free-firmware
# deb-src https://mirrors.jcut.edu.cn/debian/ ${debianName}-backports main contrib non-free non-free-firmware
deb https://mirrors.jcut.edu.cn/debian-security ${securityPath} main contrib non-free non-free-firmware
# deb-src https://mirrors.jcut.edu.cn/debian-security ${securityPath} main contrib non-free non-free-firmware`;
};

### Usage Instructions

#### Method 1

Generally, it is sufficient to replace the default repository address `deb.debian.org` with `mirrors.jcut.edu.cn` in the `/etc/apt/sources.list` file.

You can use the following command:

```shell
sudo sed -i 's|deb.debian.org|mirrors.jcut.edu.cn|g' /etc/apt/sources.list
```

Starting from *Debian 12 (bookworm)*, DEB822 format is used in container images, and the corresponding file to modify is `/etc/apt/sources.list.d/debian.sources`. You can also use the following command:

```shell
sudo sed -i 's/deb.debian.org/mirrors.jcut.edu.cn/g' /etc/apt/sources.list.d/debian.sources
```

### Notes

#### Update Index

After modifying the `sources.list` file, please run `sudo apt-get update` to update the index for the changes to take effect.

#### HTTP Pipelining Issue

In apt version 2.1.9 and later, apt's HTTP Pipelining feature appears to have a compatibility issue with Nginx servers, which may cause occasional "Connection reset by peer" errors when downloading a large number of packages from the mirror (such as system upgrades) with high bandwidth (see [Debian bug #973581](https://bugs.debian.org/cgi-bin/bugreport.cgi?bug=973581) for details).

To disable it, you can add the `-o Acquire::http::Pipeline-Depth=0` parameter when using the `apt` command, or use the following command to add the relevant settings to the apt system configuration:

```bash
echo "Acquire::http::Pipeline-Depth \"0\";" > /etc/apt/apt.conf.d/99nopipelining
```
