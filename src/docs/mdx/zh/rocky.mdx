import ConfigGenerator from '../../../components/config-generator/ConfigGenerator'

export const rockyVersionList = [
  "Rocky Linux 8",
  "Rocky Linux 9",
  "Rocky Linux 10",
];

export const defaultVersion = "Rocky Linux 9";

export const GenRockyConfig = (version) => {
  if (version === "Rocky Linux 8") {
    return `sed -e 's|^mirrorlist=|#mirrorlist=|g' \\
    -e 's|^#baseurl=http://dl.rockylinux.org/\\$contentdir|baseurl=https://mirrors.jcut.edu.cn/rocky|g' \\
    -i.bak \\
    /etc/yum.repos.d/Rocky-AppStream.repo \\
    /etc/yum.repos.d/Rocky-BaseOS.repo \\
    /etc/yum.repos.d/Rocky-Extras.repo \\
    /etc/yum.repos.d/Rocky-PowerTools.repo

# 替换完成后更新缓存
dnf makecache`;
  }
  return `sed -e 's|^mirrorlist=|#mirrorlist=|g' \\
    -e 's|^#baseurl=http://dl.rockylinux.org/\\$contentdir|baseurl=https://mirrors.jcut.edu.cn/rocky|g' \\
    -i.bak \\
    /etc/yum.repos.d/rocky-extras.repo \\
    /etc/yum.repos.d/rocky.repo

# 替换完成后更新缓存
dnf makecache`;
};

## Rocky Linux 源使用帮助

### 地址

[https://mirrors.jcut.edu.cn/rocky/](https://mirrors.jcut.edu.cn/rocky/)

### 说明

Rocky Linux 软件源

### 收录架构

x86_64, aarch64

### 收录版本

8, 9, 10

---

### 使用说明

选择您的版本，复制命令后在终端中执行即可完成换源。命令会将默认的上游地址替换为本镜像站，并保留原始配置文件的 `.bak` 备份。

<ConfigGenerator promptString="请选择您的 Rocky Linux 版本：" versionList={rockyVersionList} defaultVersion={defaultVersion} configGen={GenRockyConfig} language="bash" />

> Rocky 8 和 Rocky 9/10 的 repo 文件命名不同（前者为大写驼峰，后者为小写），请注意选择正确版本。

---

### 注意事项

以上命令只替换了默认启用的仓库。如您手动启用了其他仓库文件（如 `rocky-devel.repo`），请参照相同格式自行替换。

> 本项文档参照 http://mirrors.ustc.edu.cn/help/rocky.html 编写。
